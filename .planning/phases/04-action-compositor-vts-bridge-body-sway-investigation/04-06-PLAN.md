@on("cursor-reaction")
async def handle_cursor_reaction(ws: WebSocket, msg: dict) -> None:
    from contracts import ActionIntent, CursorReactionMessage
    payload = CursorReactionMessage.model_validate(msg)
    cursor_driver = getattr(ws.app.state, "cursor_driver", None)
    if cursor_driver is None:
        log.warning("[WS-CURSOR] cursor_driver not in app.state")
        return
    intent = ActionIntent(kind="reaction", name=payload.name, strength=payload.strength, avatar_id=payload.avatar_id)
    cursor_driver.apply_reaction_intent(
        intent,
        x=payload.x,
        y=payload.y,
        width=payload.width,
        height=payload.height,
        inside=payload.inside,
        now=__import__("time").monotonic(),
    )
```

import { useRef } from 'react'
import { send } from '@/ws/client'
import type { CursorReactionMessage } from '@contracts/ws-message'

const clamp01 = (n: number): number => Math.max(0, Math.min(1, n))

export function AvatarCursorOverlay() {
  const rafRef = useRef<number | null>(null)
  const pendingRef = useRef<CursorReactionMessage | null>(null)
  const flush = (): void => {
    rafRef.current = null
    const pending = pendingRef.current
    pendingRef.current = null
    if (pending) send(pending)
  }
  const queue = (msg: CursorReactionMessage): void => {
    pendingRef.current = msg
    if (rafRef.current === null) rafRef.current = window.requestAnimationFrame(flush)
  }
  ...
}
```
