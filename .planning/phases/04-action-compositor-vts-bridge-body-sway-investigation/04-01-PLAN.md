**Step A — `sidecar/src/sidecar/vts/pyvts_writer.py`.** Create the safe writer per RESEARCH §B verbatim, with one local adjustment: import `pyvts` (the vendored package) rather than the `pyvts_module as vts` alias the snippet shows.

```python
"""PyvtsSafeWriter — single-writer asyncio task wrapping vendored pyvts.

AVT-04 mitigation. pyvts 0.3.3 races at vts.py:117-118 (`websocket.send` then
`websocket.recv`); under `asyncio.gather` of multiple `request()` calls, the
recv() side races and sometimes raises "another coroutine is already awaiting recv()".

Mitigation pattern: one persistent _recv_loop task does all recv()s, dispatches
to per-requestID asyncio.Future map. Senders never call recv().
"""
from __future__ import annotations
import asyncio
import json
import uuid
from pathlib import Path
from typing import Any

import pyvts
from loguru import logger

from contracts import ParamFrame


class PyvtsSafeWriter:
    def __init__(
        self,
        plugin_info: dict[str, Any] | None = None,
        vts_api_info: dict[str, Any] | None = None,
        token_path: Path | None = None,
    ):
        if token_path is None:
            token_path = Path(__file__).resolve().parents[3] / ".vts_token.txt"
        if plugin_info is None:
            plugin_info = {
                "plugin_name": "AgenticLLMVTuber Compositor",
                "developer": "AgenticLLMVTuber",
                "authentication_token_path": str(token_path),
            }
        self._client = pyvts.vts(plugin_info=plugin_info, vts_api_info=vts_api_info)
        self._pending: dict[str, asyncio.Future[dict]] = {}
        self._recv_task: asyncio.Task[None] | None = None
        self._send_lock = asyncio.Lock()
        self._closed = False

    @property
    def vts_request(self):
        # Convenience: callers use writer.vts_request.requestXxx(...) just like raw pyvts
        return self._client.vts_request

    async def connect(self) -> None:
        """Open the WS and start the recv loop. Auth flow is separate (see handshake.py).

        IMPORTANT: connect() is NOT enough to send InjectParameterDataRequest — VTS
        rejects unauthenticated plugins on most write operations. Caller must follow
        with handshake.authenticate(writer).
        """
        await self._client.connect()
        self._recv_task = asyncio.create_task(self._recv_loop(), name="pyvts-recv-loop")

    async def _recv_loop(self) -> None:
        """ONLY caller of self._client.websocket.recv(). Dispatches by requestID."""
        try:
            while not self._closed:
                raw = await self._client.websocket.recv()
                msg = json.loads(raw)
                rid = msg.get("requestID")
                fut = self._pending.pop(rid, None)
                if fut is not None and not fut.done():
                    fut.set_result(msg)
                else:
                    logger.debug(f"[VTS-WRITER] unmatched response requestID={rid!r}")
        except asyncio.CancelledError:
            raise
        except Exception as exc:
            logger.warning(f"[VTS-WRITER] _recv_loop exiting: {exc!r}")
            for fut in list(self._pending.values()):
                if not fut.done():
                    fut.set_exception(exc)
            self._pending.clear()

    async def request(self, request_msg: dict, timeout: float = 5.0) -> dict:
        """Send a request; await the matching response from _recv_loop.

        Senders MUST NOT call websocket.recv() — that's the issue #51 race.
        Multiple concurrent senders are safe under this method.
        """
        request_msg = dict(request_msg)
        rid = request_msg.setdefault("requestID", uuid.uuid4().hex)
        loop = asyncio.get_running_loop()
        fut: asyncio.Future[dict] = loop.create_future()
        self._pending[rid] = fut
        async with self._send_lock:
            await self._client.websocket.send(json.dumps(request_msg))
        try:
            return await asyncio.wait_for(fut, timeout=timeout)
        except asyncio.TimeoutError:
            self._pending.pop(rid, None)
            raise

    async def inject_params(self, frame: ParamFrame) -> None:
        """Per Pitfall 4 (with deviation rationale): emits 1 + N InjectParameterDataRequest
        sends per frame, where N = len(frame.set_params).

        - 1 multi-param `requestSetMultiParameterValue(mode="add")` for the add bucket.
        - N per-param `requestSetParameterValue(mode="set", weight=...)` for the set bucket
          because pyvts's `requestSetMultiParameterValue` does NOT accept per-param weight,
          and AVT-03's intent cross-fade requires weight. Typical N = 0 in steady state
          (Path A: intent overlays go through ExpressionActivationRequest, not set_params).
          Worst case: N = 2 (joy + shy cross-fade).

        Pitfall 4's literal "≤2 sends" goal is preserved in spirit (the add bucket is
        always batched) but technically violated when set_params has multiple entries.
        Pytest test name reflects this: `test_inject_params_batches_add_and_per_param_set`.
        """
        if frame.add_params:
            msg = self._client.vts_request.requestSetMultiParameterValue(
                parameters=list(frame.add_params.keys()),
                values=list(frame.add_params.values()),
                mode="add",
            )
            await self.request(msg)
        if frame.set_params:
            # mode:"set" with weight — vts_request.requestSetMultiParameterValue
            # doesn't accept weight; fall back to per-param requestSetParameterValue
            # (set_params is small — typically ≤3 entries from the intent driver).
            for name, (value, weight) in frame.set_params.items():
                msg = self._client.vts_request.requestSetParameterValue(
                    parameter=name, value=value, weight=weight, mode="set", face_found=False,
                )
                await self.request(msg)

    async def close(self) -> None:
        self._closed = True
        if self._recv_task is not None:
            self._recv_task.cancel()
            try:
                await self._recv_task
            except (asyncio.CancelledError, Exception):
                pass
        ws = getattr(self._client, "websocket", None)
        if ws is not None:
            await self._client.close()
```

```python
"""VTS handshake — connect + auth flow.

Per Pitfall 23: must NOT block sidecar [READY] line. Caller spawns:

    handshake_task = asyncio.create_task(connect_and_authenticate(writer))

So even if the user hasn't clicked Allow in VTS yet, sidecar comes up.

from __future__ import annotations
import asyncio
from loguru import logger
from .pyvts_writer import PyvtsSafeWriter


HANDSHAKE_TIMEOUT_S = 30.0   # generous — user might be on a different monitor
RETRY_INTERVAL_S = 5.0       # auto-retry per CONTEXT Claude's discretion default


async def connect_and_authenticate(writer: PyvtsSafeWriter) -> bool:
    """Returns True on success; loops with backoff on failure."""
    while True:
        try:
            logger.info("[VTS-HANDSHAKE] connecting...")
            await writer.connect()
            logger.info("[VTS-HANDSHAKE] connected; requesting auth token (popup may appear)")
            # Use the underlying pyvts client's auth flow — it's NOT under issue #51
            # because it's a single-shot sequential operation and only runs once.
            await writer._client.request_authenticate_token()
            authenticated = await writer._client.request_authenticate()
            if authenticated:
                logger.info("[VTS-HANDSHAKE] authenticated")
                return True
            logger.warning("[VTS-HANDSHAKE] authentication failed; will retry")
        except Exception as exc:
            logger.warning(f"[VTS-HANDSHAKE] error: {exc!r}; retrying in {RETRY_INTERVAL_S}s")
        await asyncio.sleep(RETRY_INTERVAL_S)
```

```python
"""ParamID resolver — AVT-05.

For VTS renderer: write input-layer names (e.g., "ParamAngleX") and let VTS
internal routing handle smoothing via the IN-twin pattern (PROJECT_DESIGN §5.3.1).
For non-VTS renderers (post-MVP Pixi exploration per R-OPEN-2): raise
NotImplementedError — the compositor is renderer-portable but the resolver is the
single point that must be specialized for each renderer family.
"""
from __future__ import annotations
from typing import Literal


RendererName = Literal["vts", "pixi", "cubism_native"]


# VTS write-time names follow Live2D Cubism conventions; the rig's vtube.json
# routes input-layer names (e.g. ParamAngleX) to internal IN twins (ParamAngleXIN)
# automatically. We never write the IN twin directly from the compositor.
_VTS_KNOWN_INPUT_PARAMS = frozenset({
    "ParamAngleX", "ParamAngleY", "ParamAngleZ",
    "ParamEyeBallX", "ParamEyeBallY",
    "ParamEyeLOpen", "ParamEyeROpen",
    "ParamMouthOpenY", "ParamMouthForm",
    "ParamBrowLY", "ParamBrowRY",
    "ParamBodyAngleX", "ParamBodyAngleY", "ParamBodyAngleZ",
    "ParamBreath",
    # User-named params from the rig (Teto): "Lean Forward", "Auto Breath" etc.
    # We don't validate the full list here — the rig owns its parameter namespace.
})


def resolve_param_id(name: str, renderer: RendererName = "vts") -> str:
    """Resolve a logical param name to the renderer's wire-format name.

    For renderer="vts": returns the name unchanged (VTS handles IN-twin routing
    internally per PROJECT_DESIGN §5.3.1). The compositor never writes IN twins
    like "ParamAngleXIN" directly.

    For renderer="pixi" or "cubism_native": raises NotImplementedError. R-OPEN-2
    notes Pixi support is post-MVP exploratory; if/when that lands, this branch
    extends with the renderer-specific name mapping.
    """
    if renderer == "vts":
        # No-op pass-through. VTS API accepts any name the rig declares.
        return name
    if renderer in ("pixi", "cubism_native"):
        raise NotImplementedError(
            f"ParamID resolution for renderer={renderer!r} is post-MVP exploratory. "
            f"See .planning/PROJECT.md R-OPEN-2 (mobile companion via Pixi-rendering). "
            f"Skeleton ships VTS-only per AVT-05."
        )
    raise ValueError(f"Unknown renderer: {renderer!r}; expected 'vts' | 'pixi' | 'cubism_native'")
```

**Step D — `sidecar/src/sidecar/vts/discrete_dispatcher.py`.** AVT-09 path:

```python
"""DiscreteDispatcher — AVT-09 rare-trigger path.

Fires HotkeyTriggerRequest via PyvtsSafeWriter. By default refuses to fire meta
hotkeys (per teto_overrides.yaml's is_meta flag); explicit force=True bypasses.
"""
from __future__ import annotations
import asyncio
from loguru import logger

from sidecar.avatar.overrides import TetoOverrides
from contracts import DiscreteEvent
from .pyvts_writer import PyvtsSafeWriter


class DiscreteDispatcher:
    def __init__(self, writer: PyvtsSafeWriter):
        self._writer = writer

    async def fire(self, hotkey_id: str, name: str = "", force: bool = False) -> dict:
        """Fire a discrete-event hotkey by ID. Returns the VTS response."""
        msg = self._writer.vts_request.requestTriggerHotKey(hotkey_id)
        logger.info(f"[DISCRETE-DISPATCHER] firing hotkey_id={hotkey_id!r} name={name!r}")
        try:
            response = await self._writer.request(msg)
        except Exception as exc:
            logger.warning(f"[DISCRETE-DISPATCHER] fire failed: {exc!r}")
            raise
        # Emit a DiscreteEvent for any downstream consumers (logs/UI)
        loop = asyncio.get_running_loop()
        _ = DiscreteEvent(hotkey_id=hotkey_id, name=name, triggered_at=loop.time())
        return response

    async def fire_by_name(self, name: str, overrides: TetoOverrides, force: bool = False) -> dict:
        """Look up hotkey_id from teto_overrides.yaml's discovered_hotkeys and fire.

        Refuses to fire meta hotkeys (is_meta=True) unless force=True. Surfaces
        ValueError if the name is not discovered or is_meta without force.
        """
        match = next((h for h in overrides.discovered_hotkeys if h.name == name), None)
        if match is None:
            raise ValueError(
                f"Hotkey {name!r} not found in teto_overrides.discovered_hotkeys "
                f"(known: {[h.name for h in overrides.discovered_hotkeys][:5]}...)"
            )
        if match.is_meta and not force:
            raise ValueError(
                f"Hotkey {name!r} is_meta=True (e.g., Remove All Toggles) — refusing to fire. "
                f"Pass force=True to override (typically only for explicit operator commands)."
            )
        return await self.fire(match.hotkey_id, name=match.name, force=force)
```
