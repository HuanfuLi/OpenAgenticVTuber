```python
class Compositor:
    TICK_HZ = 60
    TICK_DT = 1.0 / 60
    FALL_BEHIND_THRESHOLD = 2

    def __init__(self, writer: PyvtsSafeWriter, idle_driver, speech_driver, intent_driver, cursor_driver=None):
        self._writer = writer
        self._idle, self._speech, self._intent, self._cursor = idle_driver, speech_driver, intent_driver, cursor_driver
        self._stop = False
        self._tick_count = 0
        self._dropped_frames = 0
        self._pending_strategy_swap: BodySwayStrategyName | None = None

    def request_strategy_swap(self, name: BodySwayStrategyName) -> None:
        # Pitfall 21: swap takes effect at the START of the next tick.
        self._pending_strategy_swap = name

    async def run(self) -> None:
        loop = asyncio.get_running_loop()
        start = loop.time()
        tick_n = 0
        while not self._stop:
            await self._tick(loop.time())
            tick_n += 1
            self._tick_count = tick_n
            target = start + tick_n * self.TICK_DT
            now = loop.time()
            if now > target + self.FALL_BEHIND_THRESHOLD * self.TICK_DT:
                self._dropped_frames += max(0, tick_n - int((now - start) / self.TICK_DT))
                start = now; tick_n = 0
                continue
            await asyncio.sleep(max(0.0, target - now))

    async def _tick(self, now: float) -> None:
        # Pitfall 21 gate
        if self._pending_strategy_swap is not None:
            new_name = self._pending_strategy_swap
            self._pending_strategy_swap = None
            if hasattr(self._speech, "swap_strategy"):
                self._speech.swap_strategy(new_name)

        add_acc: dict[str, float] = {}
        set_acc: dict[str, tuple[float, float]] = {}

        # 1. Idle (AVT-02 — always runs)
        for k, v in self._idle.tick(now).items():
            add_acc[k] = add_acc.get(k, 0.0) + v
        # 2. Speech
        for k, v in self._speech.tick(now).items():
            add_acc[k] = add_acc.get(k, 0.0) + v
        # 3. Intent (mode:"set" only — AVT-03 invariant)
        for k, vw in self._intent.tick(now).items():
            set_acc[k] = vw
        # 4. Cursor (mode:"add")
        if self._cursor is not None:
            for k, v in self._cursor.tick(now).items():
                add_acc[k] = add_acc.get(k, 0.0) + v

        frame = ParamFrame(add_params=add_acc, set_params=set_acc, tick_n=self._tick_count, emitted_at_monotonic=now)
        try:
            await self._writer.inject_params(frame)
        except Exception as exc:
            logger.warning(f"[COMPOSITOR] writer.inject_params failed: {exc!r}")

    async def stop(self) -> None:
        self._stop = True
```

**Step D — `exp3_modulation.py` (THE Pitfall 18 strategy):**
```python
"""exp3_modulation strategy — modulate a body-pose .exp3.json's Parameters[] by RMS.

Pitfall 18: do NOT use ExpressionActivationRequest per-tick — VTS treats it as
binary toggle and ignores fadeTime on duplicate calls. Read .exp3.json's
Parameters[] list and write each Id via InjectParameterDataRequest at strength
Value * rms_smooth. The compositor merges these into add_acc; pyvts_writer
batches them into a single InjectParameterDataRequest per tick (Pitfall 4).

.exp3.json schema (verified against Live2D/重音テト/Expressions/Star Eye.exp3.json):
  { "Type": "Live2D Expression",
    "Parameters": [{"Id": "ParamX", "Value": 1.0, "Blend": "Add" | "Multiply" | "Overwrite"}, ...] }
"""

class Exp3ModulationStrategy:
    name = "exp3_modulation"
    def __init__(self, exp3_path: Path):
        # Pitfall 18: do NOT use ExpressionActivationRequest per-tick — VTS treats it as binary toggle.
        # We load the .exp3.json once, write parameter IDs each tick via InjectParameterDataRequest.
        self._exp3_path = Path(exp3_path)
        self._param_entries = []  # list[tuple[Id, Value, Blend]]
    def init(self):
        if not self._exp3_path.exists():
            raise FileNotFoundError(f"exp3 file not found: {self._exp3_path}")
        with self._exp3_path.open(encoding="utf-8") as f:
            data = json.load(f)
        if data.get("Type") != "Live2D Expression":
            raise ValueError(f"Not a Live2D Expression file: {self._exp3_path}")
        self._param_entries = [
            (e["Id"], float(e.get("Value", 1.0)), e.get("Blend", "Add"))
            for e in data.get("Parameters", [])
        ]
    def tick(self, rms_smooth, now):
        return {pid: value * rms_smooth for pid, value, _blend in self._param_entries}
```

```python
class SpeechDriver:
    def __init__(self, speech_queue: asyncio.Queue[SpeechEnvelopePayload], overrides: TetoOverrides, avatar_dir: Path):
        self._queue = speech_queue
        self._overrides = overrides
        self._avatar_dir = avatar_dir
        self._active: SpeechEnvelopePayload | None = None
        self._rms_ema: float = 0.0
        self._strategy = build_strategy(overrides.body_sway_strategy, overrides, avatar_dir)
        self._strategy.init()

    def swap_strategy(self, name: BodySwayStrategyName) -> None:
        # Pitfall 21 mitigation: compositor calls this at tick boundary.
        # D-12: in-memory only — no write-back to teto_overrides.yaml.
        new = build_strategy(name, self._overrides, self._avatar_dir)
        new.init()
        self._strategy = new
        self._rms_ema = 0.0   # reset smoother to avoid old-strategy state bleeding through

    def _maybe_drain(self):
        try:
            self._active = self._queue.get_nowait()
        except asyncio.QueueEmpty:
            pass

    def tick(self, now: float) -> dict[str, float]:
        self._maybe_drain()
        if self._active is None or not self._active.volumes:
            return {}
        env = self._active
        slice_length = max(1, env.slice_length)
        elapsed_ms = max(0.0, (now - env.started_at) * 1000.0)
        position = elapsed_ms / slice_length
        index = int(position)
        if index >= len(env.volumes):
            self._active = None
            self._rms_ema = 0.0
            return {MOUTH_PARAM: 0.0}
        # Phase 3 D-04 linear interp
        rms_raw = env.volumes[index]
        if index + 1 < len(env.volumes):
            frac = position - index
            rms_raw = rms_raw * (1 - frac) + env.volumes[index + 1] * frac
        # EMA smoothing (RESEARCH open Q5 locked)
        self._rms_ema = EMA_ALPHA * rms_raw + (1 - EMA_ALPHA) * self._rms_ema
        out = {MOUTH_PARAM: max(0.0, min(1.0, rms_raw))}
        for k, v in self._strategy.tick(self._rms_ema, now).items():
            out[k] = v
        # D-04 investigation evidence log line
        logger.debug(
            f"[SPEECH-DRIVER strategy={self._strategy.name} rms={self._rms_ema:.3f} "
            f"mouth={out[MOUTH_PARAM]:.3f} body_params={list(out.keys())[1:]}]"
        )
        return out
```
