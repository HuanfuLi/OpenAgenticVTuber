Schema (concrete, copy-paste these field names and types):

```python
from __future__ import annotations
from pathlib import Path
from typing import List, Literal, Optional
import yaml
from pydantic import BaseModel, Field


# D-05 / smoke-pass output for one parameter probe.
class ParamProbeResult(BaseModel):
    name: str
    wrote: float
    readback: float
    visible: bool          # |readback - wrote| < 0.05
    orphan_face_tracker: bool   # |readback| < 0.05
    blend_partial: bool    # 0.05 < |readback - wrote| < 0.5


# D-11 / hotkey discovery output.
class DiscoveredHotkey(BaseModel):
    hotkey_id: str         # VTS-assigned UUID-ish
    name: str              # e.g. "Star Eye [7]"
    type: str              # e.g. "ToggleExpression"
    file: str = ""         # .exp3.json filename or "" for meta entries
    is_meta: bool = False  # True for "Remove All Toggles", "Remove Water Mark"
    llm_emittable: bool = True  # inverse of is_meta


# AVT-07 schema fields + Phase 4 D-03 additions.
BodySwayStrategyName = Literal["head_only", "proxy_param", "exp3_modulation"]


class TetoOverrides(BaseModel):
    # AVT-07 schema fields (engineer-deviation slots).
    orphan_params: List[str] = Field(default_factory=list)
    physics_chain_proxies: dict[str, str] = Field(default_factory=dict)  # source_param -> proxy_param
    sign_inversions: List[str] = Field(default_factory=list)             # params where face-tracker sign is opposite

    # D-03 body-sway strategy fields.
    body_sway_strategy: BodySwayStrategyName = "head_only"
    proxy_body_param: Optional[str] = None       # populated if proxy_param strategy is viable
    exp3_body_pose: Optional[str] = None         # path to a body-tilt .exp3.json (relative to Live2D/<rig>/Expressions/)

    # D-05 smoke-pass evidence — full per-param classification.
    param_probes: List[ParamProbeResult] = Field(default_factory=list)

    # D-11 hotkey discovery output.
    discovered_hotkeys: List[DiscoveredHotkey] = Field(default_factory=list)

    # Free-form bag for engineer notes (e.g., "VTS window title observed: 'VTube Studio'").
    notes: dict[str, str] = Field(default_factory=dict)


def load_overrides(avatar_dir: Path) -> TetoOverrides:
    """Load avatars/<id>/teto_overrides.yaml. Returns default-empty TetoOverrides if file missing
    (deliberately tolerant — sidecar boots even if smoke-pass hasn't run yet, falling back to
    body_sway_strategy='head_only' default per D-03).
    """
    yaml_path = avatar_dir / "teto_overrides.yaml"
    if not yaml_path.exists():
        return TetoOverrides()
    with yaml_path.open(encoding="utf-8") as f:
        raw = yaml.safe_load(f) or {}
    return TetoOverrides.model_validate(raw)


def save_overrides(avatar_dir: Path, overrides: TetoOverrides) -> None:
    """Write to avatars/<id>/teto_overrides.yaml as YAML — preserves field order via model_dump."""
    yaml_path = avatar_dir / "teto_overrides.yaml"
    yaml_path.parent.mkdir(parents=True, exist_ok=True)
    raw = overrides.model_dump(mode="json")
    with yaml_path.open("w", encoding="utf-8") as f:
        yaml.safe_dump(raw, f, sort_keys=False, allow_unicode=True)
```

from pathlib import Path
import pytest
from sidecar.avatar.overrides import (
    TetoOverrides, load_overrides, save_overrides,
    ParamProbeResult, DiscoveredHotkey, BodySwayStrategyName,
)


def test_default_overrides_have_head_only_strategy():
    """D-03: hardcoded compositor default body_sway_strategy is 'head_only'."""
    o = TetoOverrides()
    assert o.body_sway_strategy == "head_only"
    assert o.proxy_body_param is None

    assert o.param_probes == []

def test_load_missing_file_returns_default(tmp_path):
    """Tolerant loader: missing teto_overrides.yaml returns default TetoOverrides."""
    o = load_overrides(tmp_path)
    assert o.body_sway_strategy == "head_only"


def test_save_load_roundtrip(tmp_path):
    o = TetoOverrides(
        body_sway_strategy="proxy_param",
        proxy_body_param="Lean Forward",
        param_probes=[
            ParamProbeResult(
                name="Lean Forward", wrote=1.0, readback=0.98,
                visible=True, orphan_face_tracker=False, blend_partial=False,
            ),
        ],
        discovered_hotkeys=[
            DiscoveredHotkey(
                hotkey_id="abc123", name="Star Eye [7]",
                type="ToggleExpression", file="Star Eye.exp3.json",
                is_meta=False, llm_emittable=True,
            ),
        ],
    )
    save_overrides(tmp_path, o)
    loaded = load_overrides(tmp_path)
    assert loaded.model_dump() == o.model_dump()


def test_committed_teto_overrides_yaml_validates():
    """The committed avatars/teto/teto_overrides.yaml must validate against the schema."""
    repo_root = Path(__file__).resolve().parents[3]
    avatar_dir = repo_root / "avatars" / "teto"
    o = load_overrides(avatar_dir)
    # Strategy is one of the three valid literals
    assert o.body_sway_strategy in ("head_only", "proxy_param", "exp3_modulation")
    # If 15 hotkeys are present (smoke-pass ran), at least 2 are flagged is_meta
    if len(o.discovered_hotkeys) >= 15:
        assert sum(1 for h in o.discovered_hotkeys if h.is_meta) >= 2
        assert sum(1 for h in o.discovered_hotkeys if h.llm_emittable) >= 13
