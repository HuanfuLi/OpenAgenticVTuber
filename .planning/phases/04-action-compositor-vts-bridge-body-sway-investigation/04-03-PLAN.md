```python
"""VTS window-bounds detection (D-09 + Pitfall 20).

Pitfall 20: Unity 6000 windows use generic UnityWndClass; do NOT FindWindow by class.
Use EnumWindows + title-prefix match.

Cross-platform: on non-Windows, all functions are no-ops returning sentinel values.
AVT-10 is a Windows-only requirement per RESEARCH "Environment Availability".
"""
from __future__ import annotations
import sys
import time
from typing import Optional
from loguru import logger


_WINDOWS = sys.platform == "win32"

# Module-level cache — set on first successful EnumWindows match.
_cached_hwnd: int | None = None
_last_enum_at: float = 0.0
ENUM_REPROBE_INTERVAL_S = 30.0   # RESEARCH Pitfall 20


def find_vts_hwnd(force_reprobe: bool = False) -> int | None:
    """Returns the HWND of the VTS window, or None if not found / non-Windows."""
    global _cached_hwnd, _last_enum_at
    if not _WINDOWS:
        return None

    now = time.monotonic()
    if (not force_reprobe
        and _cached_hwnd is not None
        and (now - _last_enum_at) < ENUM_REPROBE_INTERVAL_S):
        return _cached_hwnd

    try:
        import win32gui
    except ImportError:
        logger.warning("[WINDOW-DETECT] pywin32 not installed — cursor tracker disabled")
        return None

    found: list[int] = []
    def cb(hwnd: int, _):
        if not win32gui.IsWindowVisible(hwnd):
            return True
        try:
            title = win32gui.GetWindowText(hwnd)
        except Exception:
            return True
        # Pitfall 20: title-prefix match — Unity 6000's class is generic
        if title.startswith("VTube Studio"):
            found.append(hwnd)
        return True
    try:
        win32gui.EnumWindows(cb, None)
    except Exception as exc:
        logger.warning(f"[WINDOW-DETECT] EnumWindows failed: {exc!r}")
        return None

    _last_enum_at = now
    if found:
        _cached_hwnd = found[0]
        logger.debug(f"[WINDOW-DETECT] VTS hwnd={_cached_hwnd!r}")
        return _cached_hwnd
    _cached_hwnd = None
    return None


def get_vts_rect(hwnd: int) -> tuple[int, int, int, int] | None:
    """Returns (left, top, right, bottom) in PHYSICAL pixels (DPI awareness from 04-01).
    Returns None if the call fails (e.g., VTS closed since last enum).
    """
    if not _WINDOWS or hwnd is None:
        return None
    try:
        import win32gui
        return win32gui.GetWindowRect(hwnd)
    except Exception as exc:
        logger.debug(f"[WINDOW-DETECT] GetWindowRect failed for hwnd={hwnd!r}: {exc!r}")
        return None


def get_cursor_pos() -> tuple[int, int]:
    """Returns (x, y) cursor position in PHYSICAL pixels. (0, 0) on non-Windows."""
    if not _WINDOWS:
        return (0, 0)
    try:
        import win32gui
        return win32gui.GetCursorPos()
    except Exception:
        return (0, 0)


def get_cursor_and_rect() -> tuple[tuple[int, int], tuple[int, int, int, int] | None]:
    """Convenience: one call returns (cursor_xy, vts_rect). Used by CursorDriver per tick."""
    hwnd = find_vts_hwnd()
    if hwnd is None:
        return (get_cursor_pos(), None)
    rect = get_vts_rect(hwnd)
    if rect is None:
        # VTS may have closed — invalidate cache + retry next call
        global _cached_hwnd
        _cached_hwnd = None
    return (get_cursor_pos(), rect)
```

```python
"""CursorDriver — AVT-10 + D-10.

60Hz cursor poll → ParamAngleX/Y + ParamEyeBallX/Y deflection toward cursor when
inside the VTS canvas region; cubic 800ms ease-back to neutral when cursor exits.

D-10 + RESEARCH §G:
- 80px dead-zone around face center (no cross-eyed staring).
- Linear deflection with edge clamp.
- Cubic ease-back over 800ms after cursor exit.
- Windows-only (AVT-10 explicitly Windows-scoped per RESEARCH "Environment Availability").
"""
from __future__ import annotations
from sidecar.vts.window_detect import get_cursor_and_rect
from .easing import ease_out_cubic


# D-10 constants
HEAD_MAX_DEFLECTION_DEG = 15.0
EYE_MAX_DEFLECTION_DEG = 10.0
DEAD_ZONE_PX = 80.0
EASE_BACK_DURATION_S = 0.800
FACE_CENTER_FRAC = (0.5, 0.5)   # cursor → param math centers face at (50%, 50%) of VTS rect


def _cursor_to_param_angles(
    cursor_xy: tuple[int, int],
    vts_rect: tuple[int, int, int, int],
    face_center_frac: tuple[float, float] = FACE_CENTER_FRAC,
    head_max_deg: float = HEAD_MAX_DEFLECTION_DEG,
    eye_max_deg: float = EYE_MAX_DEFLECTION_DEG,
    dead_zone_px: float = DEAD_ZONE_PX,
) -> dict[str, float]:
    """RESEARCH §G implementation. Returns {} if cursor outside vts_rect."""
    cx, cy = cursor_xy
    l, t, r, b = vts_rect
    if not (l <= cx < r and t <= cy < b):
        return {}  # cursor not in canvas → caller eases back to 0

    w = r - l
    h = b - t
    fx, fy = face_center_frac
    face_x = l + w * fx
    face_y = t + h * fy
    dx = cx - face_x
    dy = cy - face_y
    dist = (dx * dx + dy * dy) ** 0.5
    if dist < dead_zone_px:
        # D-10: no head/eye deflection inside dead-zone
        return {"ParamAngleX": 0.0, "ParamAngleY": 0.0,
                "ParamEyeBallX": 0.0, "ParamEyeBallY": 0.0}

    # Linear with clamp
    half_w = w * 0.5
    half_h = h * 0.5
    nx = max(-1.0, min(1.0, dx / half_w))
    ny = max(-1.0, min(1.0, dy / half_h))
    return {
        "ParamAngleX": nx * head_max_deg,
        "ParamAngleY": -ny * head_max_deg,    # screen-y down → param-y down
        "ParamEyeBallX": nx * eye_max_deg / head_max_deg,    # eye-gaze normalized to 1.0 max
        "ParamEyeBallY": -ny * eye_max_deg / head_max_deg,
    }


class CursorDriver:
    """Implements TickDriver Protocol from 04-02 compositor.

    State: `_last_in_canvas_output` holds the last live deflection; `_cursor_left_canvas_at`
    is None if cursor is currently in canvas, else a timestamp marking the start of ease-back.
    """
    def __init__(self):
        self._last_in_canvas_output: dict[str, float] = {}
        self._cursor_left_canvas_at: float | None = None

    def tick(self, now: float) -> dict[str, float]:
        cursor_xy, vts_rect = get_cursor_and_rect()
        if vts_rect is None:
            # VTS not running or non-Windows — no cursor contribution.
            return {}
        live = _cursor_to_param_angles(cursor_xy, vts_rect)
        if live:
            # Cursor is in canvas: live tracking, reset ease-back state.
            self._last_in_canvas_output = live
            self._cursor_left_canvas_at = None
            return live
        # Cursor exited canvas — ease back over 800ms
        if self._cursor_left_canvas_at is None:
            self._cursor_left_canvas_at = now
        elapsed = now - self._cursor_left_canvas_at
        if elapsed >= EASE_BACK_DURATION_S:
            # Ease-back complete: output rest (0); subsequent ticks return empty
            self._last_in_canvas_output = {}
            return {}
        attenuation = 1.0 - ease_out_cubic(elapsed / EASE_BACK_DURATION_S)
        return {k: v * attenuation for k, v in self._last_in_canvas_output.items()}
```
