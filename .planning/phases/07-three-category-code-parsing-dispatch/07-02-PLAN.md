```python
from __future__ import annotations

from loguru import logger
from contracts import VariantToggle
from sidecar.vts.discrete_dispatcher import DiscreteDispatcher


class VariantStateManager:
    def __init__(
        self,
        dispatcher: DiscreteDispatcher,
        reset_hotkey_id: str | None = None,
        reset_hotkey_name: str = "RemoveAllExpressions",
    ) -> None:
        self._dispatcher = dispatcher
        self._reset_hotkey_id = reset_hotkey_id
        self._reset_hotkey_name = reset_hotkey_name
        self._current_hotkey_id: str | None = None
        self._current_name: str | None = None

    @property
    def current_hotkey_id(self) -> str | None:
        return self._current_hotkey_id

    async def reset_to_baseline(self) -> None:
        if self._reset_hotkey_id:
            await self._dispatcher.fire(
                self._reset_hotkey_id,
                name=self._reset_hotkey_name,
                force=True,
            )
            logger.info("[VARIANT] reset baseline hotkey_id={}", self._reset_hotkey_id)
        else:
            logger.info("[VARIANT] reset baseline skipped; RemoveAllExpressions hotkey unavailable")
        self._current_hotkey_id = None
        self._current_name = None

    async def apply(self, toggle: VariantToggle) -> None:
        if self._current_hotkey_id == toggle.hotkey_id:
            logger.debug("[VARIANT] no-op reemit name={} hotkey_id={}", toggle.name, toggle.hotkey_id)
            return
        if self._current_hotkey_id is not None:
            await self._dispatcher.fire(
                self._current_hotkey_id,
                name=self._current_name or "",
            )
        await self._dispatcher.fire(toggle.hotkey_id, name=toggle.name)
        self._current_hotkey_id = toggle.hotkey_id
        self._current_name = toggle.name
        logger.info("[VARIANT] active name={} hotkey_id={}", toggle.name, toggle.hotkey_id)
```

Create `sidecar/src/sidecar/vts/event_completion_tracker.py`:
```python
from __future__ import annotations

import asyncio
from loguru import logger
from contracts import EventFire


class EventCompletionTracker:
    BLEND_PAD_MS = 1000
    CEILING_MS = 10000

    def __init__(self) -> None:
        self._tasks: dict[str, list[asyncio.Task]] = {}

    def in_flight_set(self) -> set[str]:
        return {hotkey_id for hotkey_id, tasks in self._tasks.items() if tasks}

    def _effective_duration_ms(self, duration_ms: int | None) -> int:
        if duration_ms is None or duration_ms <= 0 or duration_ms > self.CEILING_MS:
            return self.CEILING_MS
        return duration_ms

    def track(self, event: EventFire) -> asyncio.Task:
        total_ms = self._effective_duration_ms(event.duration_ms) + self.BLEND_PAD_MS
        task = asyncio.create_task(self._complete_after(event, total_ms / 1000.0))
        self._tasks.setdefault(event.hotkey_id, []).append(task)
        task.add_done_callback(lambda _task, hotkey_id=event.hotkey_id: self._remove_task(hotkey_id, _task))
        return task

    def _remove_task(self, hotkey_id: str, task: asyncio.Task) -> None:
        tasks = self._tasks.get(hotkey_id)
        if not tasks:
            return
        try:
            tasks.remove(task)
        except ValueError:
            return
        if not tasks:
            self._tasks.pop(hotkey_id, None)

    async def _complete_after(self, event: EventFire, delay_seconds: float) -> None:
        await asyncio.sleep(delay_seconds)
        logger.info("[EVENT-COMPLETE] hotkey_id={} name={}", event.hotkey_id, event.name)

    async def close(self) -> None:
        tasks = [task for task_list in self._tasks.values() for task in task_list]
        self._tasks.clear()
        for task in tasks:
            task.cancel()
        for task in tasks:
            try:
                await task
            except asyncio.CancelledError:
                pass
```
