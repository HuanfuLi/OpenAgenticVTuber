1. **Create `apps/renderer/src/screens/AvatarImport/usePlaceholderGate.ts`**:

   ```typescript
   /**
    * Save-disabled friction logic per CONTEXT D-A3-2.
    * Only Cubism `^exp_?\d+$` (case-insensitive) names trigger the gate.
    * VTS-extracted slugs (sv-microphone, dark-eye) and OLVT codes (hold-mic) do NOT match.
    */
   import type { VariantEntry } from '@contracts/variant-entry'

   const PLACEHOLDER_REGEX = /^exp_?\d+$/i

   export function usePlaceholderGate(variants: VariantEntry[]): {
     isDisabled: boolean
     placeholderCount: number
     firstPlaceholderIndex: number
   } {
     const indices = variants
       .map((v, i) => (PLACEHOLDER_REGEX.test(v.code) ? i : -1))
       .filter(i => i >= 0)
     return {
       isDisabled: indices.length > 0,
       placeholderCount: indices.length,
       firstPlaceholderIndex: indices[0] ?? -1,
     }
   }
   ```

   Note: pure function (no useState/useEffect) — naming convention `useFoo` is for consistency with React conventions; if linter complains, rename to `getPlaceholderGate`.

2. **Create `apps/renderer/src/screens/AvatarImport/VariantTable.tsx`** — table with 4 per-row controls per D-A3-3:

   ```tsx
   import { useState } from 'react'
   import type { VariantEntry } from '@contracts/variant-entry'
   import { COPY } from '../../lib/copy'
   import styles from './AvatarImport.module.css'

   const SLUG_VALIDATOR = /^[a-z][a-z0-9-]{0,30}$/
   const RESERVED_NAMES = new Set([
     'think', 'thinking',
     'tool_call', 'function_call', 'function_calls',
     'invoke', 'parameter',
   ])
   const PLACEHOLDER_REGEX = /^exp_?\d+$/i

   interface VariantTableProps {
     variants: VariantEntry[]
     onChange: (next: VariantEntry[]) => void
   }

   export function VariantTable({ variants, onChange }: VariantTableProps) {
     const handleEdit = (idx: number, code: string) => {
       const next = [...variants]
       next[idx] = { ...next[idx], code, is_placeholder: PLACEHOLDER_REGEX.test(code) }
       onChange(next)
     }
     const handleDelete = (idx: number) => {
       onChange(variants.filter((_, i) => i !== idx))
     }
     const validate = (code: string, idx: number): string | null => {
       if (RESERVED_NAMES.has(code.toLowerCase())) return COPY.AVATAR_IMPORT.VALIDATION_ERROR_RESERVED(code)
       const dup = variants.findIndex((v, i) => i !== idx && v.code === code)
       if (dup >= 0) return COPY.AVATAR_IMPORT.VALIDATION_ERROR_DUPLICATE(code)
       if (code && !SLUG_VALIDATOR.test(code) && !PLACEHOLDER_REGEX.test(code)) {
         return COPY.AVATAR_IMPORT.VALIDATION_ERROR_SLUG
       }
       return null
     }
     if (variants.length === 0) {
       return <p className={styles.empty}>{COPY.AVATAR_IMPORT.VARIANTS_EMPTY}</p>
     }
     return (
       <table className={styles.catalogTable} data-testid="variant-table">
         <thead>
           <tr>
             <th>{COPY.AVATAR_IMPORT.SOURCE_NAME_HEADING}</th>
             <th>{COPY.AVATAR_IMPORT.CODE_HEADING}</th>
             <th>{COPY.AVATAR_IMPORT.PREVIEW_HEADING}</th>
             <th></th>
           </tr>
         </thead>
         <tbody>
           {variants.map((v, idx) => {
             const error = validate(v.code, idx)
             const isPlaceholder = PLACEHOLDER_REGEX.test(v.code)
             return (
               <tr
                 key={`${v.hotkey_id}-${idx}`}
                 className={isPlaceholder ? styles.placeholderRow : undefined}
                 data-testid={`variant-row-${idx}`}
               >
                 <td className={styles.sourceCell}>{v.source_name}</td>
                 <td>
                   <input
                     type="text"
                     value={v.code}
                     onChange={e => handleEdit(idx, e.target.value)}
                     aria-invalid={!!error}
                     data-testid={`variant-code-input-${idx}`}
                   />
                   {error && <span className={styles.error}>{error}</span>}
                 </td>
                 <td className={styles.previewCell}>{`{${v.code}}`}</td>
                 <td>
                   <button
                     type="button"
                     onClick={() => handleDelete(idx)}
                     aria-label={COPY.AVATAR_IMPORT.DELETE_LABEL}
                     data-testid={`variant-delete-${idx}`}
                   >
                     ×
                   </button>
                 </td>
               </tr>
             )
           })}
         </tbody>
       </table>
     )
   }
   ```

1. **Create `apps/renderer/src/screens/AvatarImport/AvatarImport.tsx`** as the dedicated route. Single-page scrollable layout per D-A3-1. Concrete shape:

   ```tsx
   import { useState, useEffect, useRef } from 'react'
   import type { AvatarImportPlan, AvatarType, ImportWarning } from '@contracts/avatar-import-plan'
   import type { VariantEntry } from '@contracts/variant-entry'
   import type { EventEntry } from '@contracts/event-entry'
   import { COPY } from '../../lib/copy'
   import { setView } from '../../state/app-store'  // adapt to actual export name
   import { VariantTable } from './VariantTable'
   import { EventTable } from './EventTable'
   import { usePlaceholderGate } from './usePlaceholderGate'
   import styles from './AvatarImport.module.css'

   interface Props {
     _testInitialPlan?: AvatarImportPlan  // test-only injection; undefined in production
   }

   export function AvatarImport({ _testInitialPlan }: Props = {}) {
     const [plan, setPlan] = useState<AvatarImportPlan | null>(_testInitialPlan ?? null)
     const [variants, setVariants] = useState<VariantEntry[]>(_testInitialPlan?.variants ?? [])
     const [events, setEvents] = useState<EventEntry[]>(_testInitialPlan?.events ?? [])
     const [saving, setSaving] = useState(false)
     const [errorMsg, setErrorMsg] = useState<string | null>(null)
     const firstPlaceholderRef = useRef<HTMLTableRowElement | null>(null)

     const { isDisabled, placeholderCount, firstPlaceholderIndex } = usePlaceholderGate(variants)

     // First-launch flow: if plan is null and no test override, prompt for folder pick
     const handleImport = async () => {
       const folder = await window.api.pickAvatarFolder()
       if (!folder) return
       try {
         const newPlan = await window.api.requestImportPlan(folder)
         setPlan(newPlan)
         setVariants(newPlan.variants)
         setEvents(newPlan.events)
       } catch (e) {
         setErrorMsg(String(e))
       }
     }

     const handleSave = async () => {
       if (!plan || isDisabled) return
       setSaving(true)
       setErrorMsg(null)
       try {
         await window.api.commitAvatarOverrides({ ...plan, variants, events })
         // Success → redirect to chat
         // Toast UX: emit a console.log for now; actual toast wiring is post-MVP
         console.log(COPY.AVATAR_IMPORT.SUCCESS_TOAST)
         setView('chat')
       } catch (e) {
         setErrorMsg(String(e))
       } finally {
         setSaving(false)
       }
     }

     const scrollToFirstPlaceholder = () => {
       const row = document.querySelector(`[data-testid="variant-row-${firstPlaceholderIndex}"]`)
       row?.scrollIntoView({ behavior: 'smooth', block: 'center' })
     }

     // First-launch: no plan and no test fixture → show picker landing UI
     if (!plan) {
       return (
         <div className={styles.container}>
           <header className={styles.header}>
             <h1>{COPY.AVATAR_IMPORT.PAGE_TITLE}</h1>
             <p>{COPY.AVATAR_IMPORT.PAGE_SUBTITLE}</p>
           </header>
           <button type="button" onClick={handleImport}>
             {COPY.AVATAR_IMPORT.IMPORT_BUTTON_LABEL}
           </button>
           {errorMsg && <p className={styles.error}>{errorMsg}</p>}
         </div>
       )
     }

     // Cubism 5.3 / no-model3 → friendly error UX, no tables
     if (plan.detected_type === 'unsupported_cubism_5_3' || plan.detected_type === 'unsupported_no_model3') {
       const errorCopy = plan.detected_type === 'unsupported_cubism_5_3'
         ? COPY.AVATAR_IMPORT.ERROR_CUBISM_5_3
         : COPY.AVATAR_IMPORT.ERROR_NO_MODEL3
       return (
         <div className={styles.errorScreen}>
           <h1>{COPY.AVATAR_IMPORT.PAGE_TITLE}</h1>
           <p>{errorCopy}</p>
           <button type="button" onClick={() => setView('chat')}>
             {COPY.AVATAR_IMPORT.CANCEL_BUTTON_LABEL}
           </button>
         </div>
       )
     }

     // Main review screen: single-page scrollable per D-A3-1
     return (
       <div className={styles.container}>
         <header className={styles.header}>
           <h1>{COPY.AVATAR_IMPORT.PAGE_TITLE}</h1>
           <p>{COPY.AVATAR_IMPORT.PAGE_SUBTITLE}</p>
           <small>
             {plan.avatar_name} · {plan.detected_type} · {plan.source_rig_path}
           </small>
         </header>

         {plan.warnings.length > 0 && (
           <ul className={styles.warnings}>
             {plan.warnings.map((w: ImportWarning, i: number) => (
               <li key={i}>[{w.kind}] {w.message}</li>
             ))}
           </ul>
         )}

         <h2 className={styles.heading}>{COPY.AVATAR_IMPORT.VARIANTS_HEADING}</h2>
         <VariantTable variants={variants} onChange={setVariants} />

         <h2 className={styles.heading}>{COPY.AVATAR_IMPORT.EVENTS_HEADING}</h2>
         <EventTable events={events} onChange={setEvents} />

         <footer className={styles.footer}>
           <div>
             {isDisabled && (
               <button
                 type="button"
                 className={styles.saveDisabledMsg}
                 onClick={scrollToFirstPlaceholder}
               >
                 {COPY.AVATAR_IMPORT.SAVE_DISABLED_PLACEHOLDER(placeholderCount)}
               </button>
             )}
           </div>
           <div>
             <button type="button" onClick={() => setView('chat')} disabled={saving}>
               {COPY.AVATAR_IMPORT.CANCEL_BUTTON_LABEL}
             </button>
             <button
               type="button"
               onClick={handleSave}
               disabled={isDisabled || saving}
               aria-label="save"
             >
               {COPY.AVATAR_IMPORT.SAVE_BUTTON_LABEL}
             </button>
           </div>
         </footer>

         {errorMsg && <p className={styles.error}>{errorMsg}</p>}
       </div>
     )
   }
   ```
