   KEYBIND_TRAILING = re.compile(r'\s*\[[A-Za-z0-9]\]\s*$')
   CJK_BRACKETS_BOUNDARY = re.compile(r'【([^】]*)】')
   ASCII_SQ_BRACKET_PREFIX = re.compile(r'^\[([A-Za-z]+)\]\s*')
   SEPARATOR_CHARS = re.compile(r'[\s&＆/]+')
   SLUG_VALIDATOR = re.compile(r'^[a-z][a-z0-9-]{0,30}$')
   PLACEHOLDER_REGEX = re.compile(r'^exp_?\d+$', re.IGNORECASE)
   RESERVED_NAMES = frozenset({
       "think", "thinking",
       "tool_call", "function_call", "function_calls",
       "invoke", "parameter",
   })

   def slug_from_hotkey_name(name: str) -> str:
       s = name
       s = KEYBIND_TRAILING.sub('', s)
       s = CJK_BRACKETS_BOUNDARY.sub(r'\1-', s)        # CRITICAL: insert hyphen at CJK boundary
       s = ASCII_SQ_BRACKET_PREFIX.sub(r'\1-', s)
       s = SEPARATOR_CHARS.sub('-', s)
       s = s.lower()

       s = s.strip('-')
       if not SLUG_VALIDATOR.match(s):

       return s

   from pathlib import Path
   import json

   def read_motion3_meta(path: Path) -> tuple[float, bool]:
       """Return (duration_seconds, is_loop) from motion3.json Meta block.

       Returns (10.0, False) on missing Duration, oversized Duration (>60s),
       or unreadable file — caller resolves with PARSE-06 10s ceiling at runtime.
       """
       try:
           data = json.loads(path.read_text(encoding="utf-8"))
       except (OSError, json.JSONDecodeError):
           return (10.0, False)
       meta = data.get("Meta", {})
       duration = meta.get("Duration", 0.0)
       loop = bool(meta.get("Loop", False))
       if not isinstance(duration, (int, float)) or duration <= 0 or duration > 60:
           return (10.0, loop)
       return (float(duration), loop)

   def read_cdi3_display_names(cdi3_path: Path) -> dict[str, str]:
       """Read cdi3.json Parameters[] → dict[Id, Name].

           if pid:

       return result

       RigCapabilities, Expression, Hotkey,
   )
   from sidecar.avatar.cdi3_reader import read_cdi3_display_names
   from sidecar.avatar.overrides import AvatarOverrides

   def build_rig_capabilities(
       overrides: AvatarOverrides,
       rig_dir: Path,
   ) -> RigCapabilities:
       """Reflect RigCapabilities from rig source files + AvatarOverrides.

       Reads:
       - rig_dir/*.model3.json    Groups[].Ids → writable_param_ids
       - rig_dir/*.cdi3.json      Parameters[].Id → cdi3_display_names
       - rig_dir/*.vtube.json     ParameterSettings[].OutputLive2D → augment writable_param_ids
       - overrides.variants[]     → derive expressions
       - overrides.discovered_hotkeys[] → augment hotkeys with type info
       - overrides.sign_inversions → carry into RigCapabilities.sign_inversions
       """
       writable_ids: set[str] = set()
       # 1. model3.json Groups[].Ids
       model3 = next(rig_dir.glob("*.model3.json"), None)
       if model3 is not None:
           data = json.loads(model3.read_text(encoding="utf-8"))
           for grp in data.get("Groups", []):
               for pid in grp.get("Ids", []):
                   writable_ids.add(pid)
       # 2. .vtube.json ParameterSettings[].OutputLive2D (full param list)
       vtube = next(rig_dir.glob("*.vtube.json"), None)
       if vtube is not None:
           data = json.loads(vtube.read_text(encoding="utf-8"))
           for ps in data.get("ParameterSettings", []):
               out = ps.get("OutputLive2D")
               if out:
                   writable_ids.add(out)
       # 3. cdi3.json display names (D-A1-3: param_ranges from cdi3 only — None when absent)
       cdi3 = next(rig_dir.glob("*.cdi3.json"), None)
       cdi3_names: dict[str, str] = read_cdi3_display_names(cdi3) if cdi3 else {}
       # 4. param_ranges: cdi3.json doesn't carry numeric ranges → all None
       param_ranges: dict[str, tuple[float, float] | None] = {pid: None for pid in writable_ids}
       # 5. expressions: derive from variants[]
       expressions = [
           Expression(name=v.code, file=v.source_name)
           for v in overrides.variants
       ]
       # 6. hotkeys: derive from discovered_hotkeys (carries type info from milestone-1 smoke-pass)
       hotkeys = [
           Hotkey(name=h.name, type=h.type, hotkey_id=h.hotkey_id)
           for h in overrides.discovered_hotkeys
       ]
       return RigCapabilities(
           writable_param_ids=sorted(writable_ids),
           param_ranges=param_ranges,
           expressions=expressions,
           hotkeys=hotkeys,
           cdi3_display_names=cdi3_names,
           sign_inversions=overrides.sign_inversions,
       )

   def resolve_source_rig_path(overrides: AvatarOverrides, repo_root: Path) -> Path:
       """Resolve overrides.source_rig_path against repo_root if relative."""
       p = Path(overrides.source_rig_path)
       return p if p.is_absolute() else repo_root / p
