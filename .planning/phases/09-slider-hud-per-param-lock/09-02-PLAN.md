```typescript
import { BrowserWindow } from 'electron'
import { join } from 'node:path'
import { is } from '@electron-toolkit/utils'

let hudWindow: BrowserWindow | null = null

export function createHudWindow(): BrowserWindow {
  if (hudWindow && !hudWindow.isDestroyed()) {
    hudWindow.focus()
    return hudWindow
  }
  hudWindow = new BrowserWindow({
    width: 420,
    height: 640,
    minWidth: 360,
    minHeight: 480,
    title: 'AgenticLLMVTuber — HUD',
    show: false,
    autoHideMenuBar: true,
    webPreferences: {
      preload: join(__dirname, '../preload/index.mjs'),
      sandbox: false,
      contextIsolation: true,
      nodeIntegration: false
    }
  })

  hudWindow.on('ready-to-show', () => {
    if (hudWindow) hudWindow.show()
  })
  hudWindow.on('closed', () => {
    hudWindow = null
  })

  if (is.dev && process.env['ELECTRON_RENDERER_URL']) {
    hudWindow.loadURL(`${process.env['ELECTRON_RENDERER_URL']}#/hud`)
  } else {
    hudWindow.loadFile(join(__dirname, '../../../renderer/out/index.html'), { hash: 'hud' })
  }
  return hudWindow
}

export function closeHudWindow(): void {
  if (hudWindow && !hudWindow.isDestroyed()) {
    hudWindow.close()
  }
}

export function isHudWindowOpen(): boolean {
  return hudWindow !== null && !hudWindow.isDestroyed()
}
```

```typescript
import { useCallback, useEffect, useRef, useState } from 'react'
import type { HudMessageS2C } from '../../../../../packages/contracts/ts/hud-message-s2c'
import type { HudMessageC2S } from '../../../../../packages/contracts/ts/hud-message-c2s'

const RECONNECT_DELAY_MS = 1500
const MAX_HISTORY_FRAMES = 30  // for "Animating" filter (~2 seconds at 15Hz)

export type HudConnection = 'connecting' | 'open' | 'reconnecting'

export interface UseHudStreamResult {
  paramValues: Record<string, number>
  lockedIds: Set<string>
  connection: HudConnection
  tickN: number
  paramHistory: Record<string, number[]>
  sendSetLock: (paramId: string, value: number) => void
  sendClearLock: (paramId: string) => void
  onLocksClearedRef: { current: (() => void) | null }
}

export function useHudStream(sidecarHttpUrl: string | null): UseHudStreamResult {
  const [paramValues, setParamValues] = useState<Record<string, number>>({})
  const [lockedIds, setLockedIds] = useState<Set<string>>(new Set())
  const [connection, setConnection] = useState<HudConnection>('connecting')
  const [tickN, setTickN] = useState(0)
  const [paramHistory, setParamHistory] = useState<Record<string, number[]>>({})

  const wsRef = useRef<WebSocket | null>(null)
  const reconnectTimerRef = useRef<ReturnType<typeof setTimeout> | null>(null)
  const bufferedC2SRef = useRef<HudMessageC2S[]>([])
  const lastLockedSizeRef = useRef<number>(0)
  const onLocksClearedRef = useRef<(() => void) | null>(null)

  const send = useCallback((msg: HudMessageC2S) => {
    const ws = wsRef.current
    if (ws && ws.readyState === WebSocket.OPEN) {
      ws.send(JSON.stringify(msg))
    } else {
      bufferedC2SRef.current.push(msg)
    }
  }, [])

  const sendSetLock = useCallback((paramId: string, value: number) => {
    send({ kind: 'set-lock', param_id: paramId, value })
    setLockedIds((prev) => {
      const next = new Set(prev)
      next.add(paramId)
      return next
    })
  }, [send])

  const sendClearLock = useCallback((paramId: string) => {
    send({ kind: 'clear-lock', param_id: paramId })
    setLockedIds((prev) => {
      const next = new Set(prev)
      next.delete(paramId)
      return next
    })
  }, [send])

  useEffect(() => {
    if (!sidecarHttpUrl) return

    const wsUrl = sidecarHttpUrl.replace(/^http/, 'ws') + '/hud/ws'
    let cancelled = false

    const connect = () => {
      if (cancelled) return
      const ws = new WebSocket(wsUrl)
      wsRef.current = ws
      setConnection('connecting')

      ws.onopen = () => {
        if (cancelled) return
        setConnection('open')
        // Replay buffered set-lock/clear-lock attempts
        const buffered = bufferedC2SRef.current
        bufferedC2SRef.current = []
        for (const msg of buffered) ws.send(JSON.stringify(msg))
      }

      ws.onmessage = (ev) => {
        let msg: HudMessageS2C
        try {
          msg = JSON.parse(ev.data) as HudMessageS2C
        } catch {
          return
        }
        if (msg.kind === 'param-frame') {
          setParamValues(msg.params)
          const newLocked = new Set(msg.locked_ids)
          setLockedIds(newLocked)
          setTickN(msg.tick_n)
          setParamHistory((prev) => {
            const next = { ...prev }
            for (const [k, v] of Object.entries(msg.params)) {
              const arr = next[k] ? [...next[k], v] : [v]
              if (arr.length > MAX_HISTORY_FRAMES) arr.shift()
              next[k] = arr
            }
            return next
          })
          // D-D3 / IP-4 detection: if previous frame had locks and now empty -> avatar re-import path
          if (lastLockedSizeRef.current > 0 && msg.locked_ids.length === 0) {
            onLocksClearedRef.current?.()
          }
          lastLockedSizeRef.current = msg.locked_ids.length
        } else if (msg.kind === 'lock-confirmed') {
          // No state change; the next param-frame will reflect it. Could log for telemetry.
        } else if (msg.kind === 'lock-rejected') {
          console.error('[HUD] lock-rejected', msg.param_id, msg.reason)
        }
      }

      ws.onclose = () => {
        if (cancelled) return
        setConnection('reconnecting')
        wsRef.current = null
        reconnectTimerRef.current = setTimeout(connect, RECONNECT_DELAY_MS)
      }

      ws.onerror = () => {
        // onclose will follow; nothing to do here besides existing log
      }
    }

    connect()

    return () => {
      cancelled = true
      if (reconnectTimerRef.current) clearTimeout(reconnectTimerRef.current)
      if (wsRef.current) {
        wsRef.current.onclose = null
        wsRef.current.close()
        wsRef.current = null
      }
    }
  }, [sidecarHttpUrl])

  return {
    paramValues,
    lockedIds,
    connection,
    tickN,
    paramHistory,
    sendSetLock,
    sendClearLock,
    onLocksClearedRef
  }
}
```

```tsx
import { useCallback, useRef } from 'react'
import { Lock, Unlock } from '../../lib/icons'
import { COPY } from '../../lib/copy'

interface Props {
  paramId: string
  displayName: string | null  // from cdi3_display_names; null -> render single-line mono
  value: number
  range: [number, number]
  isLocked: boolean
  onSetLock: (paramId: string, value: number) => void
  onClearLock: (paramId: string) => void
}

const SET_LOCK_THROTTLE_MS = 33  // ~30Hz outbound
const DRAG_THRESHOLD_STEPS = 2

export function HudParamRow({
  paramId,
  displayName,
  value,
  range,
  isLocked,
  onSetLock,
  onClearLock
}: Props): JSX.Element {
  const [min, max] = range
  const span = max - min
  const step = span <= 1.0 ? 0.01 : 0.1
  const lastSendRef = useRef(0)
  const dragStartValueRef = useRef<number | null>(null)
  const isDraggingRef = useRef(false)

  const handlePointerDown = useCallback(() => {
    dragStartValueRef.current = value
    isDraggingRef.current = true
  }, [value])

  const handleChange = useCallback(
    (e: React.ChangeEvent<HTMLInputElement>) => {
      const newValue = parseFloat(e.target.value)
      if (!isDraggingRef.current) return
      const start = dragStartValueRef.current ?? newValue
      const moved = Math.abs(newValue - start) >= step * DRAG_THRESHOLD_STEPS
      if (!moved && !isLocked) return
      const now = performance.now()
      if (now - lastSendRef.current < SET_LOCK_THROTTLE_MS) return
      lastSendRef.current = now
      onSetLock(paramId, newValue)
    },
    [paramId, step, isLocked, onSetLock]
  )

  const handlePointerUp = useCallback(
    (e: React.PointerEvent<HTMLInputElement>) => {
      isDraggingRef.current = false
      const finalValue = parseFloat((e.target as HTMLInputElement).value)
      const start = dragStartValueRef.current ?? finalValue
      if (Math.abs(finalValue - start) >= step * DRAG_THRESHOLD_STEPS || isLocked) {
        onSetLock(paramId, finalValue)
      }
      dragStartValueRef.current = null
    },
    [paramId, step, isLocked, onSetLock]
  )

  const handleLockToggle = useCallback(() => {
    if (isLocked) onClearLock(paramId)
    // Unlocked + click does nothing per D-D2
  }, [isLocked, onClearLock, paramId])

  const ariaLabel = isLocked
    ? COPY.HUD.LOCK_ARIA_UNLOCK(displayName ?? paramId)
    : COPY.HUD.LOCK_ARIA_LOCK(displayName ?? paramId)

  return (
    <div className={`hud-row${isLocked ? ' locked' : ''}`} data-param-id={paramId}>
      <div className="hud-row-name">
        {displayName ? (
          <>
            <span>{displayName}</span>
            <span className="hud-row-id-sub">{paramId}</span>
          </>
        ) : (
          <span className="mono">{paramId}</span>
        )}
      </div>
      <span className="hud-row-value">{value.toFixed(2)}</span>
      <input
        className="hud-slider"
        type="range"
        min={min}
        max={max}
        step={step}
        value={value}
        onPointerDown={handlePointerDown}
        onChange={handleChange}
        onPointerUp={handlePointerUp}
        aria-label={`${displayName ?? paramId} value`}
      />
      <button
        className={`hud-lock-btn${isLocked ? ' locked' : ''}`}
        onClick={handleLockToggle}
        role="switch"
        aria-checked={isLocked}
        aria-label={ariaLabel}
        type="button"
      >
        {isLocked ? <Lock size={16} /> : <Unlock size={16} />}
      </button>
    </div>
  )
}
```

```tsx
import { useEffect, useMemo, useState } from 'react'
import type { RigCapabilities } from '../../../../../packages/contracts/ts/rig-capabilities'
import { useStore } from '../../state/app-store'
import { COPY } from '../../lib/copy'
import { HudHeader } from './HudHeader'
import { HudFilterChips, type FilterState } from './HudFilterChips'
import { HudParamRow } from './HudParamRow'
import { HudFooterStatus } from './HudFooterStatus'
import { HudErrorState } from './HudErrorState'
import { HudEmptyState } from './HudEmptyState'
import { HudLoadingState } from './HudLoadingState'
import { useHudStream } from './useHudStream'
import { ToastStack } from '../../chrome/ToastStack'

const ANIMATING_THRESHOLD = 0.01

interface RigCapsResponse extends RigCapabilities {
  hud_excluded_param_ids: string[]
}

export function HudRoot(): JSX.Element {
  const [rigCaps, setRigCaps] = useState<RigCapsResponse | null>(null)
  const [fetchError, setFetchError] = useState(false)
  const [fetchAttempt, setFetchAttempt] = useState(0)
  const [sidecarHttpUrl, setSidecarHttpUrl] = useState<string | null>(null)
  const [filters, setFilters] = useState<FilterState>({ writable: true, animating: true, locked: true })
  const { pushToast } = useStore()

  // Resolve sidecar URL once on mount
  useEffect(() => {
    if (typeof window === 'undefined' || !window.api) return
    void window.api.getReadyUrl().then((url: string | null) => {
      if (url) setSidecarHttpUrl(url)
    })
  }, [])

  const stream = useHudStream(sidecarHttpUrl)

  // Wire avatar-re-import toast (D-D3 + IP-4)
  useEffect(() => {
    stream.onLocksClearedRef.current = () => {
      pushToast({ text: COPY.HUD.TOAST_LOCKS_CLEARED })
    }
    return () => {
      stream.onLocksClearedRef.current = null
    }
  }, [stream.onLocksClearedRef, pushToast])

  // Fetch /admin/rig-capabilities
  useEffect(() => {
    if (!sidecarHttpUrl) return
    setFetchError(false)
    let cancelled = false
    const url = `${sidecarHttpUrl}/admin/rig-capabilities`
    const timeoutId = setTimeout(() => {
      if (!cancelled) setFetchError(true)
    }, 5000)
    fetch(url)
      .then((r) => {
        if (!r.ok) throw new Error(`HTTP ${r.status}`)
        return r.json() as Promise<RigCapsResponse>
      })
      .then((data) => {
        if (cancelled) return
        clearTimeout(timeoutId)
        setRigCaps(data)
      })
      .catch(() => {
        if (cancelled) return
        clearTimeout(timeoutId)
        setFetchError(true)
      })
    return () => {
      cancelled = true
      clearTimeout(timeoutId)
    }
  }, [sidecarHttpUrl, fetchAttempt])

  // Compute visible params (HUD-03 / HUD-06)
  const visibleParamIds = useMemo<string[]>(() => {
    if (!rigCaps) return []
    const excluded = new Set(rigCaps.hud_excluded_param_ids ?? [])
    return rigCaps.writable_param_ids.filter((id) => !excluded.has(id))
  }, [rigCaps])

  // Apply filters
  const filteredParamIds = useMemo<string[]>(() => {
    if (!rigCaps) return []
    return visibleParamIds.filter((id) => {
      if (filters.writable) {
        // identity filter; included for symmetry
      }
      if (filters.animating) {
        const history = stream.paramHistory[id] ?? []
        if (history.length < 2) return filters.animating ? false : true
        const min = Math.min(...history)
        const max = Math.max(...history)
        if (max - min <= ANIMATING_THRESHOLD) return false
      }
      if (filters.locked) {
        if (!stream.lockedIds.has(id)) return false
      }
      return true
    })
  }, [visibleParamIds, filters, stream.paramHistory, stream.lockedIds, rigCaps])

  // Sort alphabetic by display-name (UI-SPEC default)
  const sortedParamIds = useMemo(() => {
    if (!rigCaps) return []
    return [...filteredParamIds].sort((a, b) => {
      const an = rigCaps.cdi3_display_names[a] ?? a
      const bn = rigCaps.cdi3_display_names[b] ?? b
      return an.localeCompare(bn)
    })
  }, [filteredParamIds, rigCaps])

  // Render
  if (fetchError) {
    return (
      <div className="hud-root">
        <HudHeader connection={stream.connection} />
        <HudLoadingState error={true} onRetry={() => setFetchAttempt((n) => n + 1)} />
      </div>
    )
  }
  if (!rigCaps) {
    return (
      <div className="hud-root">
        <HudHeader connection={stream.connection} />
        <HudLoadingState error={false} onRetry={() => setFetchAttempt((n) => n + 1)} />
      </div>
    )
  }

  return (
    <div className="hud-root">
      <HudHeader connection={stream.connection} />
      <HudFilterChips filters={filters} onChange={setFilters} />
      {stream.connection === 'reconnecting' && <HudErrorState />}
      <div className="hud-list" role="list">
        {visibleParamIds.length === 0 ? (
          <HudEmptyState kind="rig" />
        ) : sortedParamIds.length === 0 ? (
          <HudEmptyState kind="filter" />
        ) : (
          sortedParamIds.map((paramId) => {
            const range = (rigCaps.param_ranges[paramId] ?? [0, 1]) as [number, number]
            const value = stream.paramValues[paramId] ?? 0
            const displayName = rigCaps.cdi3_display_names[paramId] ?? null
            return (
              <HudParamRow
                key={paramId}
                paramId={paramId}
                displayName={displayName}
                value={value}
                range={range}
                isLocked={stream.lockedIds.has(paramId)}
                onSetLock={stream.sendSetLock}
                onClearLock={stream.sendClearLock}
              />
            )
          })
        )}
      </div>
      <HudFooterStatus visible={sortedParamIds.length} locked={stream.lockedIds.size} />
      <ToastStack />
    </div>
  )
}
```
