1. **Create `sidecar/src/sidecar/avatar/import_detect.py`** per RESEARCH §"Pattern 1" lines 207-235 + §B lines 748-768. Concrete code:

   ```python
   from enum import Enum
   from pathlib import Path
   import json

   class AvatarType(str, Enum):
       OLVT = "olvt"
       VTS_STANDARD = "vts_standard"
       CUBISM_WITH_EXPRESSIONS = "cubism_with_expressions"
       CUBISM_BARE = "cubism_bare"
       UNSUPPORTED_CUBISM_5_3 = "unsupported_cubism_5_3"
       UNSUPPORTED_NO_MODEL3 = "unsupported_no_model3"

   def is_cubism_5_3_moc(moc3: Path) -> bool:
       """Read MOC3 header byte 4; True iff version >= 6 (Cubism 5.3+).

       Per RESEARCH §B: byte 4 = uint8 version field. Live2D documents
       Cubism 5.3 = version 6. VTube Studio does not yet support 5.3
       (URP rewrite ongoing per @VTubeStudio 2025-12 announcement).
       """
       try:
           with moc3.open('rb') as f:
               header = f.read(8)
       except OSError:
           return False
       if len(header) < 8 or header[:4] != b'MOC3':
           return False
       return header[4] >= 6

   def _model3_has_expressions(model3_path: Path) -> bool:
       try:
           data = json.loads(model3_path.read_text(encoding="utf-8"))
       except (OSError, json.JSONDecodeError):
           return False
       expressions = data.get("FileReferences", {}).get("Expressions", [])
       return bool(expressions)

   def detect_type(folder: Path) -> AvatarType:
       """Classify folder into one of 6 AvatarTypes.

       Order matters (per RESEARCH §Pattern 1 canonical ladder, lines 219-234):
       1. model_dict.json → OLVT (precedence wins)
       2. model3.json does NOT exist → UNSUPPORTED_NO_MODEL3 (early-exit floor)
       3. moc3 header byte 4 >= 6 → UNSUPPORTED_CUBISM_5_3
       4. *.vtube.json → VTS_STANDARD (wins over CUBISM_BARE for Teto-shape per Pitfall 1)
       5. model3.json with Expressions[] populated → CUBISM_WITH_EXPRESSIONS
       6. else → CUBISM_BARE
       """
       if (folder / "model_dict.json").exists():
           return AvatarType.OLVT
       model3 = next(folder.glob("*.model3.json"), None)
       if model3 is None:
           return AvatarType.UNSUPPORTED_NO_MODEL3
       moc3 = next(folder.glob("*.moc3"), None)
       if moc3 and is_cubism_5_3_moc(moc3):
           return AvatarType.UNSUPPORTED_CUBISM_5_3
       if any(folder.glob("*.vtube.json")):
           return AvatarType.VTS_STANDARD
       if _model3_has_expressions(model3):
           return AvatarType.CUBISM_WITH_EXPRESSIONS
       return AvatarType.CUBISM_BARE
   ```

   **CRITICAL precedence notes (per RESEARCH §Pattern 1 lines 219-234):** (1) The `model3.json is None → UNSUPPORTED_NO_MODEL3` check runs as step 2 (BEFORE moc3) — early-exit floor for empty/non-rig folders. (2) The moc3 5.3 header check runs at step 3 (AFTER confirming model3 exists, BEFORE vtube.json) so we reject Cubism 5.3 even if a sibling vtube.json is present. (3) The vtube.json check (step 4) runs BEFORE the model3.json Expressions[] check so Teto's "Cubism-bare-with-vtube-json" hybrid classifies as VTS_STANDARD per Pitfall 1.

   ```python
   import os
   import json
   from pathlib import Path
   import yaml
   import jsonschema

   _SCHEMA_PATH = Path(__file__).resolve().parents[3] / "schemas" / "avatar_overrides.schema.json"
   _SCHEMA = json.loads(_SCHEMA_PATH.read_text(encoding="utf-8"))

   def write_avatar_overrides_atomic(target: Path, data: dict) -> None:
       """Write data → target via .tmp → fsync → os.replace().

       Pre-validates against the bundled jsonschema before opening .tmp; on
       validation failure raises jsonschema.ValidationError (path-in-document
       attribute set for renderer surfacing) and does NOT create any .tmp file.

       Per CONTEXT Claude Discretion (Atomic write semantics):
       - .tmp suffix
       - os.fsync before rename
       - os.replace() (atomic on Win + POSIX since Python 3.3)
       - No .draft.yaml retained on validation failure (renderer holds draft state)
       """
       jsonschema.validate(instance=data, schema=_SCHEMA)  # raises before any FS write
       tmp = target.with_suffix(target.suffix + ".tmp")
       target.parent.mkdir(parents=True, exist_ok=True)
       yaml_text = yaml.safe_dump(data, sort_keys=False, allow_unicode=True)
       fd = os.open(tmp, os.O_WRONLY | os.O_CREAT | os.O_TRUNC, 0o644)
       try:
           os.write(fd, yaml_text.encode("utf-8"))
           os.fsync(fd)
       finally:
           os.close(fd)
       os.replace(tmp, target)  # atomic rename on Win+POSIX
   ```

   ```python
   """Avatar import HTTP endpoints — Phase 8.

   Endpoints (mounted at /admin/avatar):
   - POST /import          → detect type, run extractor, return AvatarImportPlan (no FS write)
   - POST /import/commit   → atomic-write _avatar_overrides.yaml from plan
   - GET  /import/current  → read existing _avatar_overrides.yaml for re-edit UX
   """
   from __future__ import annotations
   from pathlib import Path
   import yaml
   import jsonschema
   from fastapi import APIRouter, HTTPException
   from pydantic import BaseModel

   from sidecar.avatar.import_detect import detect_type, AvatarType
   from sidecar.avatar.extractors import (
       extract_vts, extract_cubism_named, extract_cubism_bare, extract_olvt,
   )
   from sidecar.avatar.overrides import load_avatar_overrides, AvatarOverrides, Voice
   from sidecar.avatar.overrides_writer import write_avatar_overrides_atomic
   from contracts.avatar_import_plan import AvatarImportPlan, ImportWarning

   router = APIRouter(prefix="/admin/avatar")

   # Friendly copy keys (i18n-ready strings) per RESEARCH §B lines 783-794
   COPY_ERROR_CUBISM_5_3 = (
       "This avatar uses Cubism 5.3 features that VTube Studio doesn't support yet. "
       "Please re-export the rig from Cubism Editor with target version 5.2 or earlier, "
       "or wait for VTS to add 5.3 support."
   )
   COPY_ERROR_NO_MODEL3 = (
       "This folder doesn't look like a runtime Live2D export. "
       "If you have a .cmo3 Cubism Editor project, export it first "
       "(File → Export to .moc3 in Cubism Editor)."
   )

   class ImportRequest(BaseModel):
       folder: str  # absolute path string from Electron

   class CommitResponse(BaseModel):
       status: str
       path: str

   _EXTRACTORS = {
       AvatarType.OLVT: extract_olvt,
       AvatarType.VTS_STANDARD: extract_vts,
       AvatarType.CUBISM_WITH_EXPRESSIONS: extract_cubism_named,
       AvatarType.CUBISM_BARE: extract_cubism_bare,
   }

   @router.post("/import", response_model=AvatarImportPlan)
   async def import_avatar(req: ImportRequest) -> AvatarImportPlan:
       folder = Path(req.folder)
       if not folder.exists() or not folder.is_dir():
           raise HTTPException(status_code=400, detail=f"Folder does not exist: {folder}")
       avatar_type = detect_type(folder)
       if avatar_type == AvatarType.UNSUPPORTED_CUBISM_5_3:
           return AvatarImportPlan(
               detected_type=avatar_type.value,
               source_rig_path=str(folder),
               warnings=[ImportWarning(kind="cubism_5_3", message=COPY_ERROR_CUBISM_5_3)],
           )
       if avatar_type == AvatarType.UNSUPPORTED_NO_MODEL3:
           return AvatarImportPlan(
               detected_type=avatar_type.value,
               source_rig_path=str(folder),
               warnings=[ImportWarning(kind="no_model3", message=COPY_ERROR_NO_MODEL3)],
           )
       extractor = _EXTRACTORS[avatar_type]
       variants, events, warnings = extractor(folder)
       existing = None
       overrides_path = folder / "_avatar_overrides.yaml"
       if overrides_path.exists():
           with overrides_path.open(encoding="utf-8") as f:
               raw = yaml.safe_load(f) or {}
           existing = AvatarOverrides.model_validate(raw)
       return AvatarImportPlan(
           detected_type=avatar_type.value,
           avatar_id=folder.name,
           avatar_name=folder.name,
           source_rig_path=str(folder),
           variants=variants,
           events=events,
           warnings=warnings,
           existing_overrides=existing,
       )

   @router.post("/import/commit", response_model=CommitResponse)
   async def commit_avatar(plan: AvatarImportPlan) -> CommitResponse:
       """Atomic-write _avatar_overrides.yaml from plan.

       Carries over D-A4-3 fields from existing_overrides when present:
       sign_inversions, body_sway_strategy, proxy_body_param, exp3_body_pose,
       orphan_params, physics_chain_proxies, param_probes, discovered_hotkeys,
       notes. New v2.0 fields (variants, events, voice, source_rig_path) come
       from the plan.
       """
       target_dir = Path(plan.source_rig_path)
       if not target_dir.exists():
           raise HTTPException(400, f"source_rig_path does not exist: {target_dir}")
       existing = plan.existing_overrides
       new_overrides = AvatarOverrides(
           # NEW v2.0 fields from plan
           variants=plan.variants,
           events=plan.events,
           voice=plan.voice or Voice(),
           source_rig_path=plan.source_rig_path,
           # D-A4-3 carry-overs
           sign_inversions=existing.sign_inversions if existing else [],
           body_sway_strategy=existing.body_sway_strategy if existing else "head_only",
           proxy_body_param=existing.proxy_body_param if existing else None,
           exp3_body_pose=existing.exp3_body_pose if existing else None,
           orphan_params=existing.orphan_params if existing else [],
           physics_chain_proxies=existing.physics_chain_proxies if existing else {},
           param_probes=existing.param_probes if existing else [],
           discovered_hotkeys=existing.discovered_hotkeys if existing else [],
           notes=existing.notes if existing else {},
       )
       target = target_dir / "_avatar_overrides.yaml"
       try:
           write_avatar_overrides_atomic(
               target,
               new_overrides.model_dump(mode="json", exclude_none=False),
           )
       except jsonschema.ValidationError as e:
           raise HTTPException(400, f"Validation failed: {e.message} at {list(e.path)}")
       return CommitResponse(status="ok", path=str(target))

   @router.get("/import/current", response_model=AvatarImportPlan)
   async def get_current_avatar(avatar_id: str) -> AvatarImportPlan:
       """Read existing _avatar_overrides.yaml; reformat as plan for re-edit UX.

       Looks up by avatar_id under repo-root/avatars/<id>/ first; if not found,
       returns 404. Phase 9 future-extends this to also resolve user-imported
       avatars under userData/avatars/<id>/.
       """
       # Repo-root resolution: derive from cwd or env var
       import os
       repo_root = Path(os.environ.get("AGENTICLLMVTUBER_REPO_ROOT", os.getcwd()))
       avatar_dir = repo_root / "avatars" / avatar_id
       overrides_path = avatar_dir / "_avatar_overrides.yaml"
       if not overrides_path.exists():
           raise HTTPException(404, f"No overrides for avatar_id={avatar_id}")
       overrides = load_avatar_overrides(avatar_dir)
       return AvatarImportPlan(
           detected_type="reedit",
           avatar_id=avatar_id,
           avatar_name=avatar_id,
           source_rig_path=overrides.source_rig_path,
           variants=overrides.variants,
           events=overrides.events,
           voice=overrides.voice,
           existing_overrides=overrides,
       )
   ```
