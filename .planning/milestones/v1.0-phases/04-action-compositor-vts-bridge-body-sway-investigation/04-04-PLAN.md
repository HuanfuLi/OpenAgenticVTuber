**Step B — Write `sidecar/scripts/plot_speech_evidence.py` (matplotlib plotter):**

```python
"""Plot RMS vs body-sway output from [SPEECH-DRIVER] log captures.

Usage:
    cd sidecar && uv run python -m scripts.plot_speech_evidence \\
        --input ../.planning/skeleton-verification-evidence/04/head_only/log_capture.txt \\
        --output ../.planning/skeleton-verification-evidence/04/head_only/rms_vs_output.png \\
        --primary-param ParamAngleX
"""
import argparse
import re
from pathlib import Path
import matplotlib
matplotlib.use("Agg")  # headless
import matplotlib.pyplot as plt


LOG_PATTERN = re.compile(
    r"\[SPEECH-DRIVER strategy=(?P<strat>\w+) rms=(?P<rms>[\d.]+) mouth=(?P<mouth>[\d.]+) body_params=\[(?P<bp>[^\]]*)\]"
)


def parse_log(path: Path) -> tuple[list[float], list[float], list[dict[str, float]]]:
    """Returns (timestamps_seq, rms_seq, body_params_list_of_dicts).

    Timestamps are line indices * (1/60) — exact wall-clock not needed for the plot.
    body_params dicts come from parsing the `body_params=[...]` part — adapt to your log format.
    """
    rms_series = []
    body_series = []
    with path.open(encoding="utf-8") as f:
        for line in f:
            m = LOG_PATTERN.search(line)
            if not m:
                continue
            rms_series.append(float(m.group("rms")))
            body_series.append({})  # parse `m.group("bp")` if log format has key=value pairs
    t = [i / 60.0 for i in range(len(rms_series))]
    return t, rms_series, body_series


def main():
    ap = argparse.ArgumentParser()
    ap.add_argument("--input", required=True, type=Path)
    ap.add_argument("--output", required=True, type=Path)
    ap.add_argument("--primary-param", default="ParamAngleX")
    args = ap.parse_args()
    t, rms, body = parse_log(args.input)
    fig, (ax1, ax2) = plt.subplots(2, 1, figsize=(10, 6), sharex=True)
    ax1.plot(t, rms, "b-", label="RMS smooth")
    ax1.set_ylabel("RMS")
    ax1.legend()
    primary_series = [bp.get(args.primary_param, 0.0) for bp in body]
    ax2.plot(t, primary_series, "r-", label=args.primary_param)
    ax2.set_ylabel(args.primary_param)
    ax2.set_xlabel("Time (s)")
    ax2.legend()
    plt.tight_layout()
    plt.savefig(args.output, dpi=80)
    print(f"Wrote {args.output}")


if __name__ == "__main__":
    main()
```
