    """Pydantic source-of-truth -> JSON Schema -> TypeScript codegen.

    Mechanical chain (Phase 5 plan 05-01, SC-02):
    1. Import each Pydantic model from contracts/py/contracts/.
    2. Emit JSON Schema (with our shape-mutation rules applied).
    3. Write JSON Schema to packages/contracts/generated/json-schema/{name}.schema.json (committed).
    4. Invoke `npx --yes json-schema-to-typescript` per schema.
    5. Post-process the TS output (inline hoisted aliases, dedup cross-file types, append guards).
    6. Write to packages/contracts/ts/{name}.ts (overwrites hand-written).
    """
    from __future__ import annotations
    import json, re, subprocess, sys
    from pathlib import Path

    # Make `from contracts import ...` work without installing the package.
    REPO_ROOT = Path(__file__).resolve().parents[3]
    sys.path.insert(0, str(REPO_ROOT / "packages/contracts/py"))

    from pydantic import TypeAdapter  # noqa: E402
    from contracts import (  # noqa: E402
        WSMessage, AudioPayloadMessage, ActionIntent, SpeechEnvelopePayload,
        ParamFrame, DiscreteEvent,
    )

    SCHEMA_DIR = REPO_ROOT / "packages/contracts/generated/json-schema"
    TS_DIR = REPO_ROOT / "packages/contracts/ts"
    PY_SOURCE_REL = "packages/contracts/py/contracts"

    # (model, ts_filename_without_ext, py_source_filename_without_ext, primary_type_name)
    # Order matters for cross-file dedup: emit single-type files first so cross-file
    # imports in ws-message.ts can reference already-canonical files.
    TARGETS = [
        (ActionIntent,           "action-intent",    "action_intent",   "ActionIntent"),
        (AudioPayloadMessage,    "audio-payload",    "audio_payload",   "AudioPayloadMessage"),
        (SpeechEnvelopePayload,  "speech-envelope",  "speech_envelope", "SpeechEnvelopePayload"),
        (ParamFrame,             "param-frame",      "param_frame",     "ParamFrame"),
        (DiscreteEvent,          "discrete-event",   "discrete_event",  "DiscreteEvent"),
        (WSMessage,              "ws-message",       "ws_message",      "WSMessage"),
    ]

    # Map every Pydantic class name to the TS file (without .ts) that owns its definition.
    # Used for cross-file dedup: if ws-message.ts has an interface of one of these names,
    # remove the duplicate and emit `import type` from the owning file.
    OWNER_FILE = {
        "ActionIntent":          "action-intent",
        "AudioPayloadMessage":   "audio-payload",
        "DisplayTextField":      "audio-payload",
        "SpeechEnvelopePayload": "speech-envelope",
        "ParamFrame":            "param-frame",
        "DiscreteEvent":         "discrete-event",
    }


    def force_required(schema: dict) -> None:
        """Mutate schema in place: discriminator-const + Optional[X] + defaulted fields -> required.

        Pydantic emits `default:` on every defaulted field, including the Literal
        discriminator (default = its-own-value). jsts then emits `field?:` (optional).
        Our wire contract always sends these fields; the renderer's existing TS shape
        requires them. Three rules:
          1) `const` field      -> required (discriminator literal)
          2) `anyOf` with null  -> required (Optional[X] = None always sent on wire)
          3) any `default:`     -> required (defaulted-non-null List/dict/scalar always set)
        """
        if not isinstance(schema, dict):
            return
        props = schema.get("properties")
        if isinstance(props, dict):
            required = set(schema.get("required", []))
            for name, prop in props.items():
                if not isinstance(prop, dict):
                    continue
                # Rule 1: discriminator literal.
                if "const" in prop:
                    required.add(name)
                # Rule 2: Optional[X] = None (anyOf with {"type": "null"}).
                if any(
                    isinstance(x, dict) and x.get("type") == "null"
                    for x in prop.get("anyOf", [])
                ):
                    required.add(name)
                # Rule 3: any defaulted field — wire always sends it.
                if "default" in prop:
                    required.add(name)
            if required:
                schema["required"] = sorted(required)
        # Recurse into Pydantic-generated $defs / definitions.
        for key in ("$defs", "definitions"):
            sub = schema.get(key)
            if isinstance(sub, dict):
                for v in sub.values():
                    force_required(v)


    def emit_schema(model_or_adapter, primary_name: str) -> dict:
        if hasattr(model_or_adapter, "model_json_schema"):
            schema = model_or_adapter.model_json_schema()
        else:
            schema = TypeAdapter(model_or_adapter).json_schema()
        schema["title"] = primary_name
        # additionalProperties: false at the root + every $def — prevents jsts from
        # emitting [k: string]: unknown index signatures (Pitfall 4).
        _set_additional_properties_false(schema)
        force_required(schema)
        return schema


    def _set_additional_properties_false(schema: dict) -> None:
        if not isinstance(schema, dict):
            return
        if schema.get("type") == "object" or "properties" in schema:
            schema.setdefault("additionalProperties", False)
        for key in ("$defs", "definitions"):
            sub = schema.get(key)
            if isinstance(sub, dict):
                for v in sub.values():
                    _set_additional_properties_false(v)


    def run_jsts(schema_path: Path) -> str:
        """Invoke npx json-schema-to-typescript on a schema file, return TS source."""
        result = subprocess.run(
            ["npx", "--yes", "json-schema-to-typescript",
             "--input", str(schema_path),
             "--bannerComment", "",
             "--additionalProperties", "false"],
            capture_output=True, text=True, check=True, cwd=str(REPO_ROOT),
        )
        return result.stdout


    def post_process(ts: str) -> str:
        """Inline hoisted aliases (single-literal, simple-union, primitive)."""
        # Single-string-literal-only aliases: `export type Type1 = "audio";`
        alias_re = re.compile(
            r'^export type (\w+) = ("[^"]+"(?:\s*\|\s*"[^"]+")*);$',
            re.MULTILINE,
        )
        aliases = {m.group(1): m.group(2) for m in alias_re.finditer(ts)}
        # Primitive aliases: `export type Text = string;`
        prim_re = re.compile(
            r'^export type (\w+) = (string|number|boolean|null|number\[\]|string\[\]|boolean\[\]);$',
            re.MULTILINE,
        )
        primitives = {m.group(1): m.group(2) for m in prim_re.finditer(ts)}
        # Inline references — sort by name length DESC so longer names replaced first
        # (avoids prefix collisions like Type1 vs Type11).
        for name, value in sorted({**aliases, **primitives}.items(), key=lambda x: -len(x[0])):
            ts = re.sub(rf'\b{re.escape(name)}\b', value, ts)
        # Strip dead alias declarations.
        ts = re.sub(
            r'^export type [^=]+ = "[^"]+"(?:\s*\|\s*"[^"]+")*;\n?',
            '', ts, flags=re.MULTILINE,
        )
        ts = re.sub(
            r'^export type \w+ = (?:string|number|boolean|null|\w+\[\]);\n?',
            '', ts, flags=re.MULTILINE,
        )
        # Collapse runs of 3+ blank lines to exactly 2.
        ts = re.sub(r'\n{3,}', '\n\n', ts)
        return ts


    def dedup_cross_file(ts: str, current_file_stem: str) -> tuple[str, list[str]]:
        """Strip duplicate interface declarations whose owner is another file.

        Returns (modified_ts, list_of_imports_to_prepend).
        Owner mapping is OWNER_FILE; if a class belongs to a DIFFERENT file than
        current_file_stem, remove its declaration and record the import need.
        """
        imports_needed: dict[str, list[str]] = {}  # owner_file -> [class_names]
        # Match `export interface X { ... }` blocks (single-level, non-nested).
        # Pydantic-generated interfaces are flat; jsts does not produce nested braces
        # at the top level for our contracts.
        iface_re = re.compile(
            r'^export interface (\w+) \{\n(?:[^}]|\n)*?^\}\n?',
            re.MULTILINE,
        )
        def _strip(match: re.Match) -> str:
            cls = match.group(1)
            owner = OWNER_FILE.get(cls)
            if owner and owner != current_file_stem:
                imports_needed.setdefault(owner, []).append(cls)
                return ""
            return match.group(0)
        ts = iface_re.sub(_strip, ts)
        # Build import lines (sorted for determinism).
        import_lines: list[str] = []
        for owner in sorted(imports_needed):
            classes = sorted(set(imports_needed[owner]))
            import_lines.append(
                f"import type {{ {', '.join(classes)} }} from './{owner}';"
            )
        return ts, import_lines


    def emit_guards(discriminator_mapping: dict[str, str]) -> str:
        """Generate `export const isX = (m: WSMessage): m is X => m.type === '...';`.

        discriminator_mapping comes from `schema['discriminator']['mapping']` —
        e.g., {"audio": "#/$defs/AudioPayloadMessage", ...}.
        """
        lines = ["", "// Type guards (auto-generated from discriminator mapping):"]
        for literal in sorted(discriminator_mapping):
            ref = discriminator_mapping[literal]
            variant = ref.rsplit("/", 1)[-1]  # "AudioPayloadMessage"
            suffix = variant.removesuffix("Message")  # "AudioPayload"
            lines.append(
                f"export const is{suffix} = (m: WSMessage): m is {variant} => "
                f"m.type === '{literal}';"
            )
        return "\n".join(lines) + "\n"


    def banner(py_source: str) -> str:
        return (
            f"// GENERATED FROM {PY_SOURCE_REL}/{py_source}.py — do not edit;\n"
            f"// run packages/contracts/codegen.sh to regenerate.\n\n"
        )


    def main() -> int:
        SCHEMA_DIR.mkdir(parents=True, exist_ok=True)
        TS_DIR.mkdir(parents=True, exist_ok=True)
        for model, ts_name, py_name, primary in TARGETS:
            schema = emit_schema(model, primary)
            schema_path = SCHEMA_DIR / f"{ts_name}.schema.json"
            schema_path.write_text(
                json.dumps(schema, indent=2, sort_keys=True) + "\n",
                encoding="utf-8",
            )
            ts = run_jsts(schema_path)
            ts = post_process(ts)
            ts, import_lines = dedup_cross_file(ts, ts_name)
            # Append guards to ws-message.ts only.
            if primary == "WSMessage":
                mapping = schema.get("discriminator", {}).get("mapping", {})
                ts = ts.rstrip() + "\n" + emit_guards(mapping)
            # Compose final file: banner + cross-file imports (if any) + body.
            body = ts.strip()
            import_block = "\n".join(import_lines) + "\n\n" if import_lines else ""
            final = banner(py_name) + import_block + body + "\n"
            out_path = TS_DIR / f"{ts_name}.ts"
            out_path.write_text(final, encoding="utf-8")
            print(f"[codegen] wrote {out_path.relative_to(REPO_ROOT)}")
        return 0


    if __name__ == "__main__":
        sys.exit(main())
    ```

    """Phase 5 plan 05-01 Wave 0 test (per 05-VALIDATION.md).

    """

    import os
    import shutil
    import subprocess
    import sys
    from pathlib import Path
    import pytest

    REPO_ROOT = Path(__file__).resolve().parents[3]
    TS_DIR = REPO_ROOT / "packages/contracts/ts"
    SCHEMA_DIR = REPO_ROOT / "packages/contracts/generated/json-schema"


    def _bash() -> str | None:
        # Git Bash on Windows; native bash elsewhere.
        for candidate in ("bash", "C:/Program Files/Git/bin/bash.exe"):
            if shutil.which(candidate) or Path(candidate).exists():
                return candidate
        return None


    def _venv_python() -> Path | None:
        win = REPO_ROOT / "sidecar/.venv/Scripts/python.exe"
        nix = REPO_ROOT / "sidecar/.venv/bin/python"
        if win.exists():
            return win
        if nix.exists():
            return nix
        return None


    def _read_all(d: Path) -> dict[str, bytes]:
        out: dict[str, bytes] = {}
        for p in sorted(d.glob("*")):
            if p.is_file():
                out[p.name] = p.read_bytes()
        return out


    @pytest.mark.skipif(_bash() is None, reason="bash not available (Git Bash required on Windows)")
    @pytest.mark.skipif(_venv_python() is None, reason="sidecar venv not present (run `cd sidecar && uv sync`)")
    @pytest.mark.skipif(shutil.which("npx") is None, reason="npx not on PATH")
    def test_codegen_idempotent():
        """Run codegen twice; second run produces zero-diff output."""
        bash = _bash()
        script = str(REPO_ROOT / "packages/contracts/codegen.sh")
        # First run.
        r1 = subprocess.run([bash, script], capture_output=True, text=True, cwd=str(REPO_ROOT))
        assert r1.returncode == 0, f"first codegen run failed:\nstdout:\n{r1.stdout}\nstderr:\n{r1.stderr}"
        first_ts = _read_all(TS_DIR)
        first_schema = _read_all(SCHEMA_DIR)
        # Second run.
        r2 = subprocess.run([bash, script], capture_output=True, text=True, cwd=str(REPO_ROOT))
        assert r2.returncode == 0, f"second codegen run failed:\nstdout:\n{r2.stdout}\nstderr:\n{r2.stderr}"
        second_ts = _read_all(TS_DIR)
        second_schema = _read_all(SCHEMA_DIR)
        # Compare.
        assert first_ts == second_ts, (
            f"TS output not deterministic. Diff filenames: "
            f"{sorted(set(first_ts) ^ set(second_ts))}"
        )
        assert first_schema == second_schema, (
            f"JSON Schema output not deterministic. Diff filenames: "
            f"{sorted(set(first_schema) ^ set(second_schema))}"
        )
    ```
